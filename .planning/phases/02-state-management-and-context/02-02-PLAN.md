```bash
#!/bin/bash
# Expedite SessionStart hook -- inject lifecycle context on new sessions
# Uses plain text stdout to avoid additionalContext JSON bugs (#16538, #13650, #11509)
# This is the THIRD fallback layer for context reconstruction:
#   Layer 1: !cat injection in SKILL.md (primary, fires on any /expedite:* invocation)
#   Layer 2: /expedite:status (manual, user invokes directly)
#   Layer 3: This hook (automatic, fires on session start)

STATE_FILE="$CLAUDE_PROJECT_DIR/.expedite/state.yml"

if [ ! -f "$STATE_FILE" ]; then
  exit 0  # No active lifecycle -- silent exit
fi

# Extract key fields using grep (avoids yq dependency)
PHASE=$(grep '^phase:' "$STATE_FILE" | head -1 | sed 's/phase: *"\{0,1\}\([^"]*\)"\{0,1\}/\1/')
PROJECT=$(grep '^project_name:' "$STATE_FILE" | head -1 | sed 's/project_name: *"\{0,1\}\([^"]*\)"\{0,1\}/\1/')
INTENT=$(grep '^intent:' "$STATE_FILE" | head -1 | sed 's/intent: *"\{0,1\}\([^"]*\)"\{0,1\}/\1/')

echo "Expedite lifecycle active: \"$PROJECT\" ($INTENT intent)"
echo "Current phase: $PHASE"

# Phase-aware next-step routing
case "$PHASE" in
  scope_in_progress)
    echo "Continue with /expedite:scope"
    ;;
  scope_complete)
    echo "Run /expedite:research to begin evidence gathering"
    ;;
  research_in_progress)
    echo "Continue with /expedite:research"
    ;;
  research_complete)
    echo "Run /expedite:design to generate the design"
    ;;
  research_recycled)
    echo "Re-run /expedite:research to address gaps (or use --override to proceed)"
    ;;
  design_in_progress)
    echo "Continue with /expedite:design"
    ;;
  design_complete)
    echo "Run /expedite:plan to generate the plan"
    ;;
  design_recycled)
    echo "Re-run /expedite:design to address gaps (or use --override to proceed)"
    ;;
  plan_in_progress)
    echo "Continue with /expedite:plan"
    ;;
  plan_complete)
    echo "Run /expedite:execute to begin implementation"
    ;;
  plan_recycled)
    echo "Re-run /expedite:plan to address gaps (or use --override to proceed)"
    ;;
  execute_in_progress)
    echo "Resume with /expedite:execute"
    ;;
  complete)
    echo "Lifecycle complete. Run /expedite:scope for a new lifecycle."
    ;;
  archived)
    echo "Lifecycle archived. Run /expedite:scope for a new lifecycle."
    ;;
  *)
    echo "Run /expedite:status for details"
    ;;
esac
```

**After creating the file, make it executable:**


**Key constraints:**
- Uses `$CLAUDE_PROJECT_DIR` (not `$PWD` or relative paths) to locate state.yml -- this is the platform-provided variable for the user's project root
- Uses `grep` + `sed` for field extraction -- NO `yq` or other YAML parser dependency
- Uses explicit case branches for EVERY valid phase value rather than wildcard patterns like `*_in_progress` -- this is more robust and avoids matching unexpected values
- Outputs plain text (NOT JSON additionalContext) -- this avoids platform bugs #16538, #13650, #11509
- Exits 0 silently when no state.yml exists -- no error output
- Uses `head -1` after grep to handle any duplicate field matches
- The sed pattern handles both quoted (`"value"`) and unquoted (`value`) YAML strings
- Do NOT use "arc" anywhere
- The script MUST be executable (`chmod +x`)
